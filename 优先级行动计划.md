```python
def preflight_check(args):
    """预检命令 - 检查 Story System 健康状态"""
    project_root = find_project_root()

        print("❌ 未找到项目")
```

```python
# story_system.py 添加
def health_check(self) -> Dict[str, Any]:
    """执行完整健康检查"""
    report = {
        "status": "healthy",
        "phases": {},
        "issues": [],
        "timestamp": datetime.now().isoformat()
    }
    
    # Phase 1: 检查合同种子
    report["phases"]["phase1"] = self._check_contract_seed()
    
    # Phase 2: 检查运行时合同
    report["phases"]["phase2"] = self._check_runtime_contracts()
    
    # Phase 3: 检查章节提交
    report["phases"]["phase3"] = self._check_chapter_commits()
    
    # Phase 4: 检查事件审计
    report["phases"]["phase4"] = self._check_event_audit()
    
    # Phase 5: 检查投影一致性
    report["phases"]["phase5"] = self._check_projection_consistency()
    
    return report
```

**2.3 添加数据一致性验证** (4h)
```python
def _check_projection_consistency(self) -> Dict[str, Any]:
    """检查投影一致性"""
    issues = []
    
    # 检查 state.json 与实际章节数是否一致
    state = self._load_state()
    actual_chapters = self._count_chapter_files()
    
    if state.get("total_chapters", 0) != actual_chapters:
        issues.append({
            "type": "inconsistency",
            "message": f"state.json 记录 {state.get('total_chapters')} 章，实际 {actual_chapters} 章"
        })
    
    return {"status": "ok" if not issues else "warning", "issues": issues}
```
